# Пример решения для raw-derived DDL-варианта

Вариант: `DDL-COOK. Кулинарная книга`.

## `schema.sql`

```sql
CREATE SCHEMA IF NOT EXISTS course_lab;

DROP TABLE IF EXISTS course_lab.recipe_components;
DROP TABLE IF EXISTS course_lab.recipes;
DROP TABLE IF EXISTS course_lab.products;

CREATE TABLE course_lab.recipes (
    recipe_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name text NOT NULL UNIQUE,
    description text,
    cooking_time_minutes integer NOT NULL CHECK (cooking_time_minutes > 0),
    servings integer NOT NULL CHECK (servings > 0)
);

CREATE TABLE course_lab.products (
    product_id integer GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    name text NOT NULL UNIQUE,
    price numeric(12, 2) NOT NULL CHECK (price >= 0),
    unit text NOT NULL CHECK (unit IN ('g', 'kg', 'ml', 'l', 'pcs')),
    calories_per_unit numeric(12, 2) NOT NULL CHECK (calories_per_unit >= 0)
);

CREATE TABLE course_lab.recipe_components (
    recipe_id integer NOT NULL REFERENCES course_lab.recipes(recipe_id) ON DELETE CASCADE,
    product_id integer NOT NULL REFERENCES course_lab.products(product_id),
    quantity numeric(12, 3) NOT NULL CHECK (quantity > 0),
    PRIMARY KEY (recipe_id, product_id)
);
```

## `check_constraints.sql`

```sql
TRUNCATE course_lab.recipe_components, course_lab.recipes, course_lab.products RESTART IDENTITY CASCADE;

INSERT INTO course_lab.recipes (name, description, cooking_time_minutes, servings)
VALUES ('Porridge', 'Simple breakfast', 15, 2);

INSERT INTO course_lab.products (name, price, unit, calories_per_unit)
VALUES ('Oats', 120.00, 'g', 3.8);

INSERT INTO course_lab.recipe_components (recipe_id, product_id, quantity)
VALUES (1, 1, 100);

DO $$
BEGIN
    INSERT INTO course_lab.recipes (name, cooking_time_minutes, servings)
    VALUES ('Broken', 10, 0);
    RAISE EXCEPTION 'recipe with zero servings was accepted';
EXCEPTION WHEN check_violation THEN
    RAISE NOTICE 'ok: zero servings rejected';
END $$;

DO $$
BEGIN
    INSERT INTO course_lab.products (name, price, unit, calories_per_unit)
    VALUES ('Bad product', -1, 'g', 1);
    RAISE EXCEPTION 'negative price was accepted';
EXCEPTION WHEN check_violation THEN
    RAISE NOTICE 'ok: negative price rejected';
END $$;

DO $$
BEGIN
    INSERT INTO course_lab.recipe_components (recipe_id, product_id, quantity)
    VALUES (1, 1, 50);
    RAISE EXCEPTION 'duplicate component was accepted';
EXCEPTION WHEN unique_violation THEN
    RAISE NOTICE 'ok: duplicate component rejected';
END $$;
```

## Комментарий

Этот вариант хорошо проверяет `PK`, `FK`, `UNIQUE`, `CHECK`, составной ключ и воспроизводимый проверочный сценарий. Он ближе к raw-контрольной про "Кулинарную книгу", чем основной issue-tracker сценарий.
